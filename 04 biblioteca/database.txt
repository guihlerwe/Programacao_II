create database pessoa use pessoa

CREATE TABLE Livro ( 
    Id INT AUTO_INCREMENT NOT NULL, 
    Descricao VARCHAR(100) NOT NULL, 
    Titulo VARCHAR(100) NOT NULL, 
    Autor VARCHAR(100) NOT NULL, 
    PRIMARY KEY (Id) );

CREATE TABLE Local ( 
    Id INT AUTO_INCREMENT NOT NULL, 
    Nome VARCHAR(100) NOT NULL, 
    CEP VARCHAR(100) NOT NULL, 
    Endereco VARCHAR(100) NOT NULL, 
    PRIMARY KEY (Id) );

CREATE TABLE Item ( 
    Id INT NOT NULL, 
    LocalId INT NOT NULL, 
    LivroId INT NOT NULL, 
    DataEntrada DATETIME NOT NULL, 
    PRIMARY KEY (Id) );

ALTER TABLE Item ADD CONSTRAINT livro_item_fk FOREIGN KEY (LivroId) REFERENCES Livro (Id) ON DELETE NO ACTION ON UPDATE NO ACTION;
ALTER TABLE Item ADD CONSTRAINT local_item_fk FOREIGN KEY (LocalId) REFERENCES Local (Id) ON DELETE NO ACTION ON UPDATE NO ACTION;
